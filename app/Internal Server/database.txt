sudo apt install mariadb-server
sudo nano /etc/mysql/mariadb.conf.d/50-server.cnf

changes 

bind-address = 127.0.0.1 <-> bind-address = 0.0.0.0


Create Data Base
CREATE DATABASE user_db;
USE user_db;
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    email VARCHAR(100) UNIQUE,
    password VARCHAR(255)
);


Create User
CREATE USER 'web_user'@'IP_OF_VM2' IDENTIFIED BY 'secure_password';
GRANT ALL PRIVILEGES ON user_db.* TO 'web_user'@'IP_OF_VM2';
FLUSH PRIVILEGES;

For Login 
sudo mysql -u root -p

Checking User Created or not 
SELECT user, host FROM mysql.user WHERE user = 'web_user';

For Restart 
sudo systemctl restart mariadb
